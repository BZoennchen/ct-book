\[\begin{equation*} \begin{split} b_2 + b_1 + b_0 &= a_1a_0\\ \hline 0_2 + 0_2 + 0_2 &= 00_2 \\ 0_2 + 0_2 + 1_2 &= 01_2 \\ 0_2 + 1_2 + 0_2 &= 01_2 \\ 0_2 + 1_2 + 1_2 &= 10_2 \\ 1_2 + 0_2 + 0_2 &= 01_2 \\ 1_2 + 0_2 + 1_2 &= 10_2 \\ 1_2 + 1_2 + 0_2 &= 10_2 \\ 1_2 + 1_2 + 1_2 &= 11_2 \end{split} \end{equation*}\]
